App.js::::

import React from 'react';
import './App.css';
import nav from './MyComponents/nav';
import about from './MyComponents/about';
import shop from './MyComponents/shop';
import {BrowserRouter as Router, Switch, Route} from 'react-router-dom';
function App(){
    return (
      <Router>
      <div className="App">
          <h1>MyReactApp</h1>
          <nav />
          <switch>
          <Route path="/" exact component={Home} />
          <Route path="/MyComponents/about" component={about}/>
          <Route path="/MyComponents/shop" component={shop}/>
          </switch>
      </div>
      </Router>
    ); 
}

const Home = () => (
  <div>
    <h1>Home Page</h1>
  </div>
)

export default App;
